import string
import cv2
import numpy as np
import easyocr
from easyocr import Reader
reader=easyocr.Reader(['en'])

frameWidth = 640    #Frame Width
franeHeight = 480   # Frame Height

plateCascade = cv2.CascadeClassifier("D:\hack\haarcascade_russian_plate_number.xml")
minArea = 500

cap =cv2.VideoCapture(0)
cap.set(3,frameWidth)
cap.set(4,franeHeight)
cap.set(10,150)
count = 0


success=cv2.imread("D:\hack\images\i3.jpg")
img  = cv2.imread("D:\hack\images\i3.jpg")

imgGray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

numberPlates = plateCascade.detectMultiScale(imgGray, 1.1, 4)

for (x, y, w, h) in numberPlates:
        area = w*h
        if area > minArea:

            imgRoi = img[y:y+h,x:x+w]
            break


test=reader.readtext(imgRoi)
if len(test)!=0:
    print(test[0][-2])
    print("accuracy= ",test[0][-1])
